function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const tasksSheet = ensureSheet(ss, "Tasks", [ "TaskID","Kid","TaskType","Subject","Title","StartPage","EndPage","PointPerUnit","DueDate","Approved","ManualComplete","CompletedAt","Deleted","CreatedAt" ]); const checksSheet = ensureSheet(ss, "PageChecks", [ "TaskID","Page","Done","Date" ]); const rewardsSheet = ensureSheet(ss, "Rewards", [ "RewardID","Kid","Point","Memo","Approved","Date" ]); const tasks = rowsToObjects(tasksSheet); const checks = rowsToObjects(checksSheet); const rewards = rowsToObjects(rewardsSheet); const latestCheckMap = {}; checks.forEach(c => { const taskId = String(c.TaskID || c.ID || c.id); const page = String(c.Page); if (!taskId || !page) return; latestCheckMap[taskId + "::" + page] = String(c.Done).toLowerCase() !== "false"; }); const tasksWithDone = tasks .filter(t => String(t.Deleted).toLowerCase() !== "true") .map(t => { const id = t.TaskID || t.ID || t.id; const donePages = Object.keys(latestCheckMap) .filter(key => key.startsWith(String(id) + "::") && latestCheckMap[key] === true) .map(key => Number(key.split("::")[1])); return { id: id, kid: t.Kid, type: t.TaskType || t.Type || "page", subject: t.Subject, title: t.Title, start: Number(t.StartPage || 1), end: Number(t.EndPage || 1), point: Number(t.PointPerUnit || t.PointPerPage || t.XPPerPage || 1), dueDate: formatDateValue(t.DueDate), approved: String(t.Approved).toLowerCase() === "true", manualComplete: String(t.ManualComplete).toLowerCase() === "true", completedAt: formatDateValue(t.CompletedAt), done: [...new Set(donePages)] }; }).filter(t => t.id && t.kid && t.title); const rewardData = rewards.map(r => ({ id: r.RewardID, kid: r.Kid, point: Number(r.Point || r.XP || 0), memo: r.Memo, approved: String(r.Approved).toLowerCase() === "true", date: formatDateValue(r.Date) })).filter(r => r.id && r.kid); return ContentService .createTextOutput(JSON.stringify({ tasks: tasksWithDone, rewards: rewardData })) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); const action = data.action; const payload = data.payload; const tasksSheet = ensureSheet(ss, "Tasks", [ "TaskID","Kid","TaskType","Subject","Title","StartPage","EndPage","PointPerUnit","DueDate","Approved","ManualComplete","CompletedAt","Deleted","CreatedAt" ]); const checksSheet = ensureSheet(ss, "PageChecks", [ "TaskID","Page","Done","Date" ]); const rewardsSheet = ensureSheet(ss, "Rewards", [ "RewardID","Kid","Point","Memo","Approved","Date" ]); if (action === "addTask") { appendByHeaders(tasksSheet, { TaskID: payload.id, Kid: payload.kid, TaskType: payload.type || "page", Subject: payload.subject, Title: payload.title, StartPage: payload.start || 1, EndPage: payload.end || 1, PointPerUnit: payload.point || 1, DueDate: payload.dueDate || "", Approved: false, ManualComplete: false, CompletedAt: "", Deleted: false, CreatedAt: new Date() }); } if (action === "togglePage") { checksSheet.appendRow([payload.id, payload.page, payload.done !== false, new Date()]); } if (action === "editTask") { updateManyById(tasksSheet, "TaskID", payload.id, { Subject: payload.subject, Title: payload.title, StartPage: payload.start, EndPage: payload.end, PointPerUnit: payload.point, DueDate: payload.dueDate || "" }); } if (action === "completeTask") { updateManyById(tasksSheet, "TaskID", payload.id, { ManualComplete: true, CompletedAt: payload.completedAt || new Date() }); if (payload.done && payload.done.length) { payload.done.forEach(p => checksSheet.appendRow([payload.id, p, true, new Date()])); } } if (action === "reopenTask") { updateManyById(tasksSheet, "TaskID", payload.id, { ManualComplete: false, CompletedAt: "" }); } if (action === "deleteTask" || action === "rejectTask") { updateManyById(tasksSheet, "TaskID", payload.id, {Deleted: true}); } if (action === "requestReward") { appendByHeaders(rewardsSheet, { RewardID: payload.id, Kid: payload.kid, Point: payload.point || 0, Memo: payload.memo || "現金に交換", Approved: false, Date: new Date() }); } if (action === "approveTask") updateById(tasksSheet, "TaskID", payload.id, "Approved", true); if (action === "approveReward") updateById(rewardsSheet, "RewardID", payload.id, "Approved", true); if (action === "rejectReward") deleteById(rewardsSheet, "RewardID", payload.id); return ContentService .createTextOutput(JSON.stringify({ result: "ok" })) .setMimeType(ContentService.MimeType.JSON); } function ensureSheet(ss, name, headers) { let sheet = ss.getSheetByName(name); if (!sheet) sheet = ss.insertSheet(name); const lastCol = Math.max(sheet.getLastColumn(), headers.length); const currentHeaders = sheet.getRange(1, 1, 1, lastCol).getValues()[0]; if (currentHeaders.every(h => h === "")) { sheet.getRange(1, 1, 1, headers.length).setValues([headers]); return sheet; } const existing = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; headers.forEach(h => { if (!existing.includes(h)) { sheet.getRange(1, sheet.getLastColumn() + 1).setValue(h); existing.push(h); } }); return sheet; } function rowsToObjects(sheet) { if (!sheet) return []; const values = sheet.getDataRange().getValues(); if (values.length < 2) return []; const headers = values[0]; return values.slice(1).filter(row => row.some(v => v !== "")).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); } function appendByHeaders(sheet, obj) { const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const row = headers.map(h => Object.prototype.hasOwnProperty.call(obj, h) ? obj[h] : ""); sheet.appendRow(row); } function formatDateValue(value) { if (!value) return ""; if (Object.prototype.toString.call(value) === "[object Date]") { return Utilities.formatDate(value, Session.getScriptTimeZone(), "yyyy-MM-dd"); } return String(value); } function updateById(sheet, idHeader, id, targetHeader, value) { const obj = {}; obj[targetHeader] = value; updateManyById(sheet, idHeader, id, obj); } function updateManyById(sheet, idHeader, id, updates) { const values = sheet.getDataRange().getValues(); const headers = values[0]; const idCol = headers.indexOf(idHeader) + 1; if (idCol <= 0) return; for (let r = 2; r <= values.length; r++) { if (String(sheet.getRange(r, idCol).getValue()) === String(id)) { Object.keys(updates).forEach(h => { const col = headers.indexOf(h) + 1; if (col > 0) sheet.getRange(r, col).setValue(updates[h]); }); return; } } } function deleteById(sheet, idHeader, id) { const values = sheet.getDataRange().getValues(); const headers = values[0]; const idCol = headers.indexOf(idHeader) + 1; if (idCol <= 0) return; for (let r = values.length; r >= 2; r--) { if (String(sheet.getRange(r, idCol).getValue()) === String(id)) sheet.deleteRow(r); } }
